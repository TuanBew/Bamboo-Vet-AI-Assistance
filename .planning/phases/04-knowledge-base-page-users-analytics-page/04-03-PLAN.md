```typescript
export interface KhachHangFilters {

export interface KhachHangData {
  new_by_month: Array<{ month: string; count: number }>   // LineChart: "2024-01" format
  by_province: Array<{ name: string; count: number }>     // BarChart
  by_district: Array<{ name: string; count: number }>     // Horizontal BarChart

  all_customers: {
    kpis: {
      active_count: number     // Con hoat dong
      mapped_pct: number       // Da phan tuyen %
      geo_pct: number          // Da dinh vi %
      type_count: number       // So loai cua hieu (distinct types with count > 0)

    breakdown: CustomerBreakdown[]  // 8 rows, one per type
  }

  purchasing_customers: {
    kpis: {
      total_with_orders: number
      mapped_pct: number
      geo_pct: number
      type_count: number
    }
    breakdown: PurchasingBreakdown[]
  }

  high_value_stores: Array<{
    customer_code: string
    customer_name: string
    customer_type: string
    province: string
    total_value: number
  }>
}
```

**Implementation of `getKhachHangData(filters: KhachHangFilters): Promise<KhachHangData>`:**

1. `const db = createServiceClient()`

2. Fetch all customers:
   ```typescript
   const { data: customers } = await db.from('customers').select('*')
   ```

3. Fetch all customer_purchases:
   ```typescript
   const { data: purchases } = await db.from('customer_purchases').select('customer_id, total_value')
   ```

4. **new_by_month**: Group customers by `created_at` month (YYYY-MM format), count per month. Sort ascending.

5. **by_province**: Group customers by `province`, count per province. Sort descending by count. Take top 15.

6. **by_district**: Group customers by `district`, count per district. Sort descending by count. Take top 15.

7. **all_customers**:
   - `active_count`: count where `is_active = true`
   - `mapped_pct`: `(count is_mapped=true / total) * 100`, round to 1 decimal
   - `geo_pct`: `(count is_geo_located=true / total) * 100`, round to 1 decimal
   - `type_count`: distinct `customer_type` values with count > 0
   - `breakdown`: for each of 8 types (TH, GSO, PHA, SPS, BTS, OTHER, PLT, WMO), count customers, compute `pct = count/total * 100`.
   - Type label mapping: TH="Tap hoa", GSO="Bach hoa", PHA="Nha thuoc", SPS="Me & Be", BTS="My pham", OTHER="Khac", PLT="Phu lieu toc", WMO="Cho".

8. **purchasing_customers**:
   - Find customers with at least 1 purchase: `Set<customer_id>` from purchases.
   - For those customers: compute same KPIs (total_with_orders, mapped_pct, geo_pct, type_count).
   - Breakdown: same per-type count, PLUS `pct_of_total = count / all_customers_total * 100` and `pct_of_active = count / all_customers_active * 100`.
   - NOTE: `total_with_orders` counts unique customers with orders, NOT total purchase events. The reference image showing 588 > 455 reflects order event counts, so also compute purchase event counts if needed. But for the KPI tile, use the total number of purchase records (not distinct customers) to match the reference image's intent.

9. **high_value_stores**:
   - Group purchases by customer_id, SUM(total_value).
   - Filter where sum > 300000.
   - Join with customer info.
   - Return array sorted descending by total_value.

**Create `app/api/admin/khach-hang/route.ts`:**

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { requireAdmin } from '@/lib/admin/auth'
import { getKhachHangData } from '@/lib/admin/services/khach-hang'

export async function GET(request: NextRequest) {
  const auth = await requireAdmin()
  if (auth instanceof NextResponse) return auth

  const { searchParams } = new URL(request.url)
  const npp = searchParams.get('npp') || ''

  try {
    const data = await getKhachHangData({ npp })
    return NextResponse.json(data)
  } catch (error) {
    console.error('Khach hang API error:', error)
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 })
  }
}
```
