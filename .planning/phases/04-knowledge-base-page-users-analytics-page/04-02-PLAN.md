```typescript
import { createServiceClient } from '@/lib/supabase/server'

export interface TonKhoFilters {
  snapshot_date: string  // YYYY-MM-DD, default today
  nhom: string           // product_group filter, empty = all

export interface TonKhoData {
  kpis: {
    total_value: number       // SUM(qty * unit_price) for latest snapshots
    total_qty: number         // SUM(qty)

    total_sku: number         // total product count
  }
  value_by_nhom: Array<{ name: string; value: number }>         // Gia tri ton theo nhom
  value_by_brand: Array<{ name: string; value: number }>        // Gia tri ton theo thuong hieu (classification)
  value_by_category: Array<{ name: string; value: number }>     // Gia tri ton theo nganh hang
  qty_by_nhom: Array<{ name: string; value: number }>           // So luong ton theo nhom
  qty_by_brand: Array<{ name: string; value: number }>          // So luong ton theo thuong hieu
  qty_by_category: Array<{ name: string; value: number }>       // So luong ton theo nganh hang
  products: Array<{
    product_code: string
    product_name: string
    qty: number
    min_stock: number         // hardcoded default 10

  }>
```

**Implementation of `getTonKhoData(filters: TonKhoFilters): Promise<TonKhoData>`:**

1. `const db = createServiceClient()`

2. Fetch latest snapshots per product (Approach A from research):
   ```typescript
   const { data: snapshots } = await db
     .from('inventory_snapshots')
     .select('product_id, snapshot_date, qty, unit_price')
     .lte('snapshot_date', filters.snapshot_date)
     .order('snapshot_date', { ascending: false })

   // Deduplicate: keep only latest snapshot per product
   const latestByProduct = new Map<string, typeof snapshots[0]>()
   for (const snap of snapshots ?? []) {
     if (!latestByProduct.has(snap.product_id)) {
       latestByProduct.set(snap.product_id, snap)
     }
   }
   ```

3. Fetch products for grouping info:
   ```typescript
   const { data: products } = await db
     .from('products')
     .select('id, product_code, product_name, product_group, classification, manufacturer')
   ```

4. Fetch last import dates from purchase_order_items:
   ```typescript
   const { data: orderItems } = await db
     .from('purchase_order_items')
     .select('product_id, purchase_orders!inner(order_date)')
     .order('purchase_orders(order_date)', { ascending: false })
   // Group by product_id, take first (latest) order_date
   ```
   If this join syntax does not work with Supabase, fall back to:
   ```typescript
   const { data: orderItems } = await db
     .from('purchase_order_items')
     .select('product_id, quantity')
   const { data: orders } = await db
     .from('purchase_orders')
     .select('id, order_date')
   // Join in JS, group by product_id, take MAX order_date
   ```

5. Join snapshots with product info in JS. Apply `nhom` filter (if non-empty, filter by `product_group`). Apply `search` filter (case-insensitive match on product_code or product_name).

6. Compute KPIs from filtered latest snapshots:
   - `total_value = SUM(qty * unit_price)`
   - `total_qty = SUM(qty)`
   - `sku_in_stock = COUNT where qty > 0`
   - `total_sku = total filtered products count`

7. Compute chart aggregations by grouping:
   - `value_by_nhom`: group by `product_group`, SUM(qty * unit_price)
   - `value_by_brand`: group by `classification`, SUM(qty * unit_price)
   - `value_by_category`: group by `manufacturer`, SUM(qty * unit_price)
   - Same 3 groupings for qty (SUM qty only)
   - Sort each array descending by value.

8. Build products list with: product_code, product_name, qty, min_stock (hardcode 10), last_import_date (from step 4, or 'N/A'), unit_price, total_value (qty * unit_price).

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { requireAdmin } from '@/lib/admin/auth'
import { getTonKhoData } from '@/lib/admin/services/ton-kho'

export async function GET(request: NextRequest) {
  const auth = await requireAdmin()
  if (auth instanceof NextResponse) return auth

  const { searchParams } = new URL(request.url)
  const snapshot_date = searchParams.get('snapshot_date') || new Date().toISOString().slice(0, 10)
  const nhom = searchParams.get('nhom') || ''
  const search = searchParams.get('search') || ''

  try {
    const data = await getTonKhoData({ snapshot_date, nhom, search })
    return NextResponse.json(data)
  } catch (error) {
    console.error('Ton kho API error:', error)
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 })
  }
}
```
