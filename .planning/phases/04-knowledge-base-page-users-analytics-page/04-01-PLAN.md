```typescript
function groupBy<T>(items: T[], key: keyof T, sumKey?: keyof T): Array<{ name: string; count: number }> {
  const map = new Map<string, number>()
  for (const item of items) {
    const k = String(item[key])
    const v = sumKey ? Number(item[sumKey]) : 1
    map.set(k, (map.get(k) || 0) + v)
  }
  return Array.from(map.entries())
    .map(([name, count]) => ({ name, count }))
    .sort((a, b) => b.count - a.count)
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { requireAdmin } from '@/lib/admin/auth'
import { getKnowledgeBaseData } from '@/lib/admin/services/knowledge-base'

export async function GET(request: NextRequest) {
  const auth = await requireAdmin()
  if (auth instanceof NextResponse) return auth

  const { searchParams } = new URL(request.url)
  const page = Number(searchParams.get('page')) || 1
  const page_size = Number(searchParams.get('page_size')) || 10
  const search = searchParams.get('search') || ''
  const doc_type = searchParams.get('doc_type') || ''
  const category = searchParams.get('category') || ''

  try {
    const data = await getKnowledgeBaseData({ page, page_size, search, doc_type, category })
    return NextResponse.json(data)
  } catch (error) {
    console.error('Knowledge Base API error:', error)
    return NextResponse.json({ error: 'Internal server error' }, { status: 500 })
  }
}
```

Key details:
- `requireAdmin()` is called first — non-admin gets 403
- Query params: `page` (default 1), `page_size` (default 10), `search` (default ''), `doc_type` (optional), `category` (optional)
- Returns JSON response from getKnowledgeBaseData
- Wraps in try/catch with 500 error response
