3. Create `lib/pdf/vietnamese-font.ts` with this structure:
```typescript
/**
 * Roboto Regular TTF font encoded as base64 for jsPDF Vietnamese support.
 * License: Apache License 2.0 (Google Fonts)
 *
 * This file is dynamically imported only when PDF export is triggered,
 * so it does NOT increase the initial page load bundle size.
 */
import type { jsPDF } from 'jspdf'

export const ROBOTO_REGULAR_BASE64 = '...' // paste the full base64 string here

/**
 * Register Roboto font with a jsPDF document instance.
 * Call this BEFORE any text rendering or autoTable call.
 */
export function addVietnameseFont(doc: jsPDF): void {
  doc.addFileToVFS('Roboto-Regular.ttf', ROBOTO_REGULAR_BASE64)
  doc.addFont('Roboto-Regular.ttf', 'Roboto', 'normal')
  doc.setFont('Roboto')
}
```

The base64 string will be ~230KB of text. This is acceptable because the file is only loaded via dynamic import when the user clicks the PDF export button.

```typescript
const handlePdf = useCallback(async () => {
  const jsPDFModule = await import('jspdf')
  const jsPDF = jsPDFModule.default
  await import('jspdf-autotable')
  const { addVietnameseFont } = await import('@/lib/pdf/vietnamese-font')

  const doc = new jsPDF({ orientation: 'landscape', unit: 'mm', format: 'a4' })
  addVietnameseFont(doc)

  const headers = columns.map(c => c.label)
  const body = data.map(row =>
    columns.map(c => {
      const val = row[c.key]
      return val == null ? '' : String(val)
    })
  )

  ;(doc as unknown as Record<string, Function>).autoTable({
    head: [headers],
    body,
    startY: 15,
    styles: { fontSize: 7, cellPadding: 2, font: 'Roboto' },
    headStyles: { fillColor: [31, 41, 55], textColor: [255, 255, 255] },
    theme: 'grid',
  })

  doc.save('export.pdf')
}, [data, columns])
```
