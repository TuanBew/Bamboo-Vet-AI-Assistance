```typescript
async function batchInsert(table: string, rows: Record<string, unknown>[], chunkSize = 500) {
  console.log(`  Inserting ${rows.length} rows into ${table}...`)
  for (let i = 0; i < rows.length; i += chunkSize) {
    const chunk = rows.slice(i, i + chunkSize)
    const { error } = await supabase.from(table).upsert(chunk, { onConflict: 'id', ignoreDuplicates: true })
    if (error) {
      console.error(`  Error inserting chunk ${i}-${i + chunk.length} into ${table}:`, error.message)
      // Continue with next chunk rather than failing entirely
    }
    if (i > 0 && i % 5000 === 0) {
      console.log(`  ... ${i}/${rows.length} rows inserted`)
    }
  }
  console.log(`  Done: ${rows.length} rows into ${table}`)
}
```
