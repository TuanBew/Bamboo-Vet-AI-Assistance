4. Create an inner `MapController` component that uses `useMap()` hook and calls `map.flyTo([lat, lng], zoom)`. This component is placed inside `MapContainer` as a child. It receives a ref via `useImperativeHandle` to expose the `flyTo` method.
5. Wrap the default export with `forwardRef<MapHandle, MapViewProps>`.
6. Use `useImperativeHandle` to expose `flyTo` from the `MapController` ref.
7. Keep all existing pin rendering, marker icons, etc. unchanged.
8. Export the `MapHandle` type.

The key pattern:
```typescript
import { forwardRef, useImperativeHandle, useRef } from 'react'
import { MapContainer, TileLayer, Marker, Popup, useMap } from 'react-leaflet'

export type MapHandle = { flyTo: (lat: number, lng: number, zoom?: number) => void }

function FlyToController({ mapRef }: { mapRef: React.MutableRefObject<MapHandle | null> }) {
  const map = useMap()
  mapRef.current = {
    flyTo: (lat, lng, zoom = 15) => map.flyTo([lat, lng], zoom)
  }
  return null
}

const LeafletMapInner = forwardRef<MapHandle, MapViewProps>(function LeafletMapInner(props, ref) {
  const internalRef = useRef<MapHandle | null>(null)
  useImperativeHandle(ref, () => ({
    flyTo: (lat, lng, zoom) => internalRef.current?.flyTo(lat, lng, zoom)
  }))
  // ... rest of component with <FlyToController mapRef={internalRef} /> inside MapContainer
})
export default LeafletMapInner
```
