```sql
-- suppliers
CREATE TABLE IF NOT EXISTS suppliers (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  supplier_code text NOT NULL UNIQUE,
  supplier_name text NOT NULL,
  province      text,
  created_at    timestamptz NOT NULL DEFAULT now()
);

-- products
CREATE TABLE IF NOT EXISTS products (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  product_code     text NOT NULL UNIQUE,
  product_name     text NOT NULL,
  product_group    text NOT NULL,
  classification   text NOT NULL,
  packaging        text,
  manufacturer     text,
  unit_price       numeric(12,0) NOT NULL,
  created_at       timestamptz NOT NULL DEFAULT now()
);

-- purchase_orders
CREATE TABLE IF NOT EXISTS purchase_orders (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  order_code    text NOT NULL UNIQUE,
  order_date    date NOT NULL,
  supplier_id   uuid NOT NULL REFERENCES suppliers(id) ON DELETE RESTRICT,
  total_amount  numeric(15,0) NOT NULL,
  total_promo_qty integer NOT NULL DEFAULT 0,
  created_at    timestamptz NOT NULL DEFAULT now()
);

-- purchase_order_items
CREATE TABLE IF NOT EXISTS purchase_order_items (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  order_id    uuid NOT NULL REFERENCES purchase_orders(id) ON DELETE CASCADE,
  product_id  uuid NOT NULL REFERENCES products(id) ON DELETE RESTRICT,
  quantity    integer NOT NULL,
  promo_qty   integer NOT NULL DEFAULT 0,
  unit_price  numeric(12,0) NOT NULL,
  subtotal    numeric(15,0) GENERATED ALWAYS AS (quantity * unit_price) STORED,
  created_at  timestamptz NOT NULL DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_poi_order_id ON purchase_order_items (order_id);
CREATE INDEX IF NOT EXISTS idx_poi_product_id ON purchase_order_items (product_id);
```

**Export function `getNhapHangData(filters: NhapHangFilters): Promise<NhapHangData>`**

Implementation using Supabase JS client (no raw SQL):

1. **Fetch suppliers list** (always, for dropdown): `supabase.from('suppliers').select('id, supplier_name')`

2. **Fetch orders for month/npp**: Query `purchase_orders` with `.gte('order_date', startOfMonth).lte('order_date', endOfMonth)`. If npp provided, add `.eq('supplier_id', npp)`. Join with suppliers to get supplier_name: use a separate query or fetch supplier names from the already-fetched list.

3. **Fetch all order items for matched orders**: Get order IDs from step 2, then query `purchase_order_items` with `.in('order_id', orderIds)`. Join products to get product details: fetch products separately and join in JS.

4. **Compute KPIs from items:**
   - total_revenue = SUM(quantity * unit_price) across all items
   - total_quantity = SUM(quantity)
   - total_promo_qty = SUM(promo_qty)
   - total_orders = orders.length
   - total_skus = COUNT(DISTINCT product_id)
   - avg_per_order = total_revenue / total_orders (or 0 if no orders)

5. **Compute daily_revenue:** Group items by order date's day, sum revenue per day.

6. **Compute daily_quantity:** Group items by order date's day, sum quantity and promo_qty per day.

7. **Build order_items map:** Key by order_code, value is array of items with product details. This MUST be a `Record<string, Array<{...}>>` keyed by order_code — not an empty object or flat array. For each order, look up its items and map them with product name, code, quantity, promo_qty, unit_price, and subtotal (quantity * unit_price).

8. **Compute top_products:** Group by product, sum revenue, sort DESC, take top 10.

9. **Compute by_industry:** Group by product_group, sum revenue.

10. **Compute by_product_group:** Group by classification, sum revenue.

11. **Compute by_brand:** Group by manufacturer, sum revenue.

**CRITICAL — order_date format transformation:** Supabase returns `order_date` as ISO `'YYYY-MM-DD'` (dash format). The spec mandates `'YYYY/MM/DD'` (slash format). When building the `orders` array, transform each order_date: `po.order_date.replace(/-/g, '/')` to convert from ISO dash to spec-required slash format.

All numeric fields parsed with `Number()` to avoid string issues from Supabase.
