```typescript
import { cn } from '@/lib/utils'

export interface KpiCardProps {
  value: string | number
  label: string
  icon?: React.ReactNode
  bgColor?: string        // Tailwind bg class, e.g. "bg-blue-500"
  textColor?: string      // Tailwind text class, default "text-white"
  className?: string
}

export function KpiCard({ value, label, icon, bgColor = 'bg-gray-800', textColor = 'text-white', className }: KpiCardProps) {
  return (
    <div className={cn(
      'rounded-xl p-4 flex flex-col gap-1',
      bgColor,
      textColor,
      className
    )}>
      {icon && <div className="mb-1">{icon}</div>}
      <span className="text-2xl font-bold">{value}</span>
      <span className="text-sm opacity-80">{label}</span>
    </div>
  )
}
```

This is a simple presentational component. No 'use client' needed unless icon is a ReactNode that requires client features — but since it receives icon as a prop, no client directive needed. If build fails, add 'use client'.

Properties match the spec: colored card with large number, label, optional icon, optional background color prop.

```typescript
'use client'

import { useState } from 'react'
import { ChevronDown, ChevronRight } from 'lucide-react'
import { cn } from '@/lib/utils'

export interface SectionHeaderProps {
  title: string
  defaultOpen?: boolean
  children: React.ReactNode
  className?: string
}

export function SectionHeader({ title, defaultOpen = true, children, className }: SectionHeaderProps) {
  const [isOpen, setIsOpen] = useState(defaultOpen)

  return (
    <div className={cn('mb-6', className)}>
      <button
        onClick={() => setIsOpen(!isOpen)}
        className="flex items-center gap-2 w-full text-left px-4 py-2 bg-teal-600/20 rounded-lg mb-3"
      >
        {isOpen ? (
          <ChevronDown className="h-4 w-4 text-teal-400" />
        ) : (
          <ChevronRight className="h-4 w-4 text-teal-400" />
        )}
        <span className="text-sm font-semibold text-teal-400 uppercase tracking-wider">
          {title}
        </span>
      </button>
      {isOpen && <div>{children}</div>}
    </div>
  )
}
```

The SectionHeader renders a teal header bar with title and collapsible chevron toggle. Uses `useState` for collapse state (hence 'use client'). The teal color matches the sidebar section labels for visual consistency. Animation timing is at Claude's discretion (using simple show/hide for now — no transition animation).
